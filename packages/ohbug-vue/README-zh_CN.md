## 安装

```
pnpm instal @ohbug/browser @ohbug/vue
```

## 使用

```javascript
import Vue from 'vue'
import Ohbug from '@ohbug/browser'
import OhbugVue from '@ohbug/vue'

const client = Ohbug.setup({ apiKey: 'YOUR_API_KEY' })

Vue.createApp(App)
  .use(OhbugVue(client))
  .mount('#app')
```
